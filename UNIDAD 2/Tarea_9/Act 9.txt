Table Pacientes {
  id_paciente int [pk, increment]
  nombre varchar(100)
  apellido_paterno varchar(100)
  apellido_materno varchar(100)
  fecha_nacimiento date
  sexo varchar(20)
  telefono varchar(20)
  correo varchar(100)
  direccion varchar(200)
}

Table Especialidades {
  id_especialidad int [pk, increment]
  nombre varchar(100)
  descripcion varchar(200)
}

Table Medicos {
  id_medico int [pk, increment]
  nombre varchar(100)
  apellido_paterno varchar(100)
  apellido_materno varchar(100)
  telefono varchar(20)
  correo varchar(100)
  id_especialidad int
}

Table Citas {
  id_cita int [pk, increment]
  fecha date
  hora time
  motivo_consulta varchar(200)
  estado varchar(50)
  id_paciente int
  id_medico int
}

Table Diagnosticos {
  id_diagnostico int [pk, increment]
  descripcion text
  fecha_diagnostico date
  id_cita int
}

Table Tratamientos {
  id_tratamiento int [pk, increment]
  descripcion text
  fecha_inicio date
  fecha_fin date
  id_diagnostico int
}

Table Medicamentos {
  id_medicamento int [pk, increment]
  nombre varchar(100)
  descripcion varchar(200)
  dosis varchar(100)
}

Table Recetas {
  id_receta int [pk, increment]
  fecha_receta date
  indicaciones text
  id_cita int
}

Table Receta_Medicamento {
  id_receta int
  id_medicamento int
  cantidad varchar(50)
  frecuencia varchar(100)
}

Table Signos_Vitales {
  id_signo int [pk, increment]
  fecha_registro datetime
  peso decimal(5,2)
  estatura decimal(5,2)
  temperatura decimal(4,2)
  presion_arterial varchar(20)
  frecuencia_cardiaca int
  id_paciente int
}

Ref: Diagnosticos.id_cita > Citas.id_cita

Ref: Tratamientos.id_diagnostico > Diagnosticos.id_diagnostico

Ref: Recetas.id_cita > Citas.id_cita

Ref: Receta_Medicamento.id_receta > Recetas.id_receta
Ref: Receta_Medicamento.id_medicamento > Medicamentos.id_medicamento

Ref: Signos_Vitales.id_paciente > Pacientes.id_paciente
